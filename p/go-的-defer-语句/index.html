	ID int
}

func NewConn(id int) *Conn {
	return &Conn{ID: id}
}

func (c *Conn) Close() error {
	fmt.Printf("close %d!\n", c.ID)
	return nil
}

func main() {
	arr := make([]Conn, 5)
	for i := range arr {
		arr[i].ID = i
	}

	for _, conn := range arr {
		defer conn.Close()
	}
}
